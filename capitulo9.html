Capítulo 9: Módulos e Bibliotecas

9.1 Introdução aos Módulos

Módulos são arquivos Python que contêm definições e declarações. Eles permitem organizar código em arquivos separados e reutilizar funcionalidades.

Vantagens dos Módulos

  • Organização: Separa código em arquivos lógicos
  • Reutilização: Use o mesmo código em diferentes projetos
  • Namespace: Evita conflitos de nomes
  • Manutenção: Facilita correções e atualizações
# Importando módulos da biblioteca padrão import math import random import datetime # Usando funções dos módulos print(f"Raiz quadrada de 16: {math.sqrt(16)}") print(f"Número aleatório: {random.randint(1, 100)}") print(f"Data atual: {datetime.datetime.now()}") # Importando funções específicas from math import pi, cos, sin from random import choice from datetime import date print(f"\nValor de π: {pi}") print(f"cos(π): {cos(pi)}") print(f"sin(π/2): {sin(pi/2)}") print(f"Escolha aleatória: {choice(['Python', 'Java', 'JavaScript'])}") print(f"Data de hoje: {date.today()}") # Importando com alias (apelido) import datetime as dt import math as matematica agora = dt.datetime.now() print(f"\nHora atual: {agora.strftime('%H:%M:%S')}") print(f"Valor de e: {matematica.e}")

Você pode criar seus próprios módulos salvando código Python em arquivos .py e importando-os em outros scripts.

9.4 Pacotes (Packages)

Pacotes são diretórios que contêm múltiplos módulos. Eles ajudam a organizar projetos grandes em uma estrutura hierárquica.

# Estrutura de um pacote # meu_projeto/ # __init__.py # matematica/ # __init__.py # basica.py # avancada.py # utilidades/ # __init__.py # texto.py # arquivo.py # Conteúdo de matematica/basica.py def somar(a, b): return a + b def subtrair(a, b): return a - b # Conteúdo de matematica/avancada.py import math def logaritmo(x, base=math.e): return math.log(x, base) def seno(angulo_graus): return math.sin(math.radians(angulo_graus)) # Conteúdo de matematica/__init__.py from .basica import somar, subtrair from .avancada import logaritmo, seno __version__ = "1.0.0" __author__ = "Curso Python" # Conteúdo de utilidades/texto.py def capitalizar_palavras(texto): return ' '.join(palavra.capitalize() for palavra in texto.split()) def contar_palavras(texto): return len(texto.split()) def inverter_texto(texto): return texto[::-1] # Como usar o pacote from meu_projeto.matematica import somar, logaritmo from meu_projeto.utilidades.texto import capitalizar_palavras # Ou importar o pacote completo import meu_projeto.matematica as mat import meu_projeto.utilidades.texto as txt resultado = mat.somar(5, 3) texto_formatado = txt.capitalizar_palavras("python é incrível")

# Módulo: gerenciador_arquivos.py """ Módulo Gerenciador de Arquivos Fornece funcionalidades para manipular arquivos e diretórios. Autor: Curso Python Versão: 1.0 """ import os import shutil from pathlib import Path from datetime import datetime import json class GerenciadorArquivos: """Classe para gerenciar operações com arquivos e diretórios.""" def __init__(self, diretorio_base="."): self.diretorio_base = Path(diretorio_base) self.historico = [] def listar_arquivos(self, diretorio=None, extensao=None): """ Lista arquivos em um diretório. Args: diretorio (str): Diretório a listar (padrão: diretório base) extensao (str): Filtrar por extensão (ex: '.txt') Returns: list: Lista de arquivos """ if diretorio is None: diretorio = self.diretorio_base else: diretorio = Path(diretorio) if not diretorio.exists(): raise FileNotFoundError(f"Diretório não encontrado: {diretorio}") arquivos = [] for item in diretorio.iterdir(): if item.is_file(): if extensao is None or item.suffix.lower() == extensao.lower(): info = { 'nome': item.name, 'caminho': str(item), 'tamanho': item.stat().st_size, 'modificado': datetime.fromtimestamp(item.stat().st_mtime) } arquivos.append(info) return sorted(arquivos, key=lambda x: x['nome']) def criar_diretorio(self, nome_diretorio): """ Cria um novo diretório. Args: nome_diretorio (str): Nome do diretório a criar Returns: bool: True se criado com sucesso """ caminho = self.diretorio_base / nome_diretorio try: caminho.mkdir(parents=True, exist_ok=True) self._adicionar_historico(f"Diretório criado: {caminho}") return True except Exception as e: print(f"Erro ao criar diretório: {e}") return False def copiar_arquivo(self, origem, destino): """ Copia um arquivo. inicio self._registrar_requisicao('DELETE', url, response.status_code, tempo_resposta) if response.status_code in [200, 204]: print(f"✅ DELETE {endpoint} - {response.status_code} ({tempo_resposta:.2f}s)") return True else: print(f"❌ DELETE {endpoint} - Autor: Curso Python Versão: 1.0 """ import json import os from pathlib import Path from configparser import ConfigParser from typing import Any, Dict, Optional class ConfigManager: """Gerenciador de configurações com suporte a múltiplos formatos.""" def __init__(self, config_dir="config", ambiente="development"): self.config_dir = Path(config_dir) self.ambiente = ambiente self.config_data = {} self.config_dir.mkdir(exist_ok=True) #
